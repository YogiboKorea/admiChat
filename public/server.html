배너 클릭 히트맵 (관리자 대시보드)

실제 웹사이트 방문자가 클릭한 메인 배너(Alt 기준)의 분포도를 기간/기기별로 확인합니다.

~
💻📱 누적 클릭: 0회
👆 상단에서 메뉴를 선택하고 "히트맵 보기"를 클릭해주세요.
(배너 클릭 데이터가 수집되면 위 목록에 즉시 표시됩니다.)
👉 카페24 쇼핑몰 연동 스크립트 보기 (클릭) 

<script>
document.addEventListener('DOMContentLoaded', function() {
    // 백엔드 API 주소 (환경에 맞게 변경)
    const API_BASE_URL = 'https://port-0-admichat-lzgmwhc4d9883c97.sel4.cloudtype.app';

    const visualArea = document.querySelector('.visual-area');
    if (!visualArea) return;

    visualArea.addEventListener('click', async function(e) {
        const slide = e.target.closest('.swiper-slide');
        if (!slide) return;
        
        const img = slide.querySelector('img');
        if (!img) return;

        // PC/모바일 영역 구분 (반응형 화면 대응)
        let deviceType = 'pc';
        // 1순위: 클릭된 노드가 모바일 래퍼(#visual_m) 안에 있으면 모바일
        if (e.target.closest('#visual_m') !== null) {
            deviceType = 'mobile';
        } else if (window.innerWidth <= 1200) {
            // 2순위: 사용자 CSS 분기점(1200px) 기준 모바일로 간주
            deviceType = 'mobile';
        }

        const rect = img.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;

        if (x < 0 || y < 0 || x > rect.width || y > rect.height) return;

        const bannerAlt = img.getAttribute('alt') || '이름_없는_배너';
        let imageUrl = img.getAttribute('src') || '';
        if (imageUrl.startsWith('/')) {
            imageUrl = window.location.origin + imageUrl; // 상대 경로를 절대 경로로 변경 (어드민 볼 때 엑스박스 방지)
        }
        
        const linkElem = slide.querySelector('a');
        const linkUrl = linkElem ? linkElem.getAttribute('href') : window.location.href;

        const clickData = {
            bannerId: bannerAlt,       
            imageUrl: imageUrl,        
            url: linkUrl,
            deviceType: deviceType,
            x: Math.round(x),
            y: Math.round(y),
            screenWidth: rect.width,
            screenHeight: rect.height
        };

        try {
            fetch(`${API_BASE_URL}/api/track-click`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(clickData)
            });
        } catch(err) { console.error('히트맵 통신 오류', err); }
        });
    });
});
</script>
